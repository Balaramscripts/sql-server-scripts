select database_name, name,
(backup_size / (1024 * 1024 * 1024)) AS BackupSizeGB,
(compressed_backup_size / (1024 * 1024 * 1024)) AS CompressedBackupSizeGB
FROM msdb.dbo.backupset b
JOIN msdb.dbo.backupmediafamily m ON b.media_set_id = m.media_set_id
WHERE b.type = 'D'AND 
backup_finish_date BETWEEN '2024-08-25 00:00:00' AND '2024-08-25 23:59:59'
ORDER BY backup_finish_date DESC
