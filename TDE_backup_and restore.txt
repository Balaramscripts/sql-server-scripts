USE master;
GO

SELECT db.name AS DatabaseName,
       dm.encryption_state,
       CONVERT(VARCHAR(40), dm.encryptor_thumbprint, 2) AS encryptor_thumbprint,
       c.name AS CertificateName, 
       CONVERT(VARCHAR(40), c.thumbprint, 2) AS cert_thumbprint
FROM sys.dm_database_encryption_keys dm
JOIN sys.databases db ON dm.database_id = db.database_id
LEFT JOIN sys.certificates c ON dm.encryptor_thumbprint = c.thumbprint;
GO
